function setVacation(id){

  var vacationSubtableId = 3005798;

  var mainQuery = db.getAPIQuery("/ragicannual-leave/20001"); //取得員工清單
  var mainEntry = mainQuery.getAPIEntry(id);

  var gender = mainEntry.getFieldValue(3000956); //員工性別

  var vacationParamQuery = db.getAPIQuery("/ragicannual-leave/20004"); //取得給假參考
  var vacationParamResults = vacationParamQuery.getAPIResultList();
  var vacationParamEntry; //符合時間的參考資料
  var vacationSubtableSize; //該年假期的種類數量

  for(var i = 0; i < vacationParamResults.length; i ++){

    var endDate = new Date(entry.getFieldValue(3005805));

    if(nowDate.getTime() >= startDate.getTime() && nowDate.getTime() <= endDate.getTime()) {
      vacationParamEntry = entry;
      vacationSubtableSize = entry.getSubtableSize(3005813);
      break;
    }
  }

  for(var i = 0; i < vacationSubtableSize; i ++) {
    var allowedGenders = vacationParamEntry.getSubtableFieldValue(3005813, i, 3005807); // 假期限定性別
    var vacationName = vacationParamEntry.getSubtableFieldValue(3005813, i, 3005806); // 假期種類名稱

    var isGenderAllowed = checkGender(gender, allowedGenders);

    // 跳過不符合條件的假別
    if (!isGenderAllowed) {
      continue;
    }

    if(vacationName === "特休") {
      fixedLeaveHours = 0;
    }

    mainEntry.setSubtableFieldValue(3005790, -i-1, vacationName);
    mainEntry.setSubtableFieldValue(3005791, -i-1, nowDate.getFullYear());

}



function padZero(num) {
  return num < 10 ? '0' + num : num.toString();
}

function checkGender(gender, allowedGenders) {
  var isGenderAllowed = false;
  if (allowedGenders === '通用') {
    isGenderAllowed = true; // 不限性別，所有人都允許
  } else if (allowedGenders === '限女性' && gender === '女') {
    isGenderAllowed = true; // 限女性，且員工性別為女
  } else if (allowedGenders === '限男性' && gender === '男') {
    isGenderAllowed = true; // 限男性，且員工性別為男
  }
  return isGenderAllowed;
}
